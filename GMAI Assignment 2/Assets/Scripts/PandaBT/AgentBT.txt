tree("Root")
	repeat
		fallback
			tree("Approached")
			tree("Attending")
			tree("Idle")

tree("Idle")
	sequence
		EnterIdleState
		ChooseRandomLocation

tree("Approached")
	sequence
		CheckApproach
		MoveToCounter

tree("Attending")
	sequence
		CheckAttending
		fallback
			sequence
				CheckCustomerInquired
				tree("PotionInquiry")
			sequence
				CheckCustomerLeft
				tree("Idle")

tree("PotionInquiry")
	sequence
		InitializeInquiry
		repeat
			fallback
				sequence
					CheckHealingOption
					tree("Transaction")
				sequence
					CheckArcaneOption
				sequence
					CheckThirdOption

tree("Transaction")
	sequence
		InitializeTransaction
		CheckPaid
